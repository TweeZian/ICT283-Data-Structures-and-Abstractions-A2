487 template<class T>
488 void Bst<T>::DeleteFromTree(Node<T>* &p)
489 {
490  Node<T> *current; //pointer to traverse the tree
491  Node<T> *trailCurrent; //pointer behind current
492  Node<T> *temp; //pointer to delete the node
493 
494  if (p == nullptr) //check if node to delete is empty
495  {
496  cout << "Error: The node to be deleted does not exist." << endl;
497  }
498 
499  //check node is leaf node
500  else if (p->left == nullptr && p->right == nullptr)
501  {
502  temp = p;
503  p = nullptr;
504  delete temp;
505  }
506  //check node has no left child (has right)
507  else if (p->left == nullptr)
508  {
509  temp = p;
510  p = temp->right; //keep right child using p
511  delete temp;
512  }
513  //check node has no right child (has left)
514  else if (p->right == nullptr)
515  {
516  temp = p;
517  p = temp->left; //keep left child using p
518  delete temp;
519  }
520  else
521  {
522  //this method uses the DELETE node's left child most bottom
523  //right node to replace the current node position, as the
524  //bottom node will always be bigger than everything in the
525  //left child tree but smaller than the node's right
526  current = p->left; //start from node's left
527  trailCurrent = nullptr;
528 
529  //move down left child's right till bottom
530  while (current->right != nullptr)
531  {
532  trailCurrent = current;
533  current = current->right;
534  }
535 
536  //replace deleted node with bottom most right node
537  p->info = current->info;
538 
539  //current did not move, left child of delete node has no
540  //right node (bottomed out)
541  if (trailCurrent == nullptr)
542  {
543  //move delete node's left child up one level to
544  //take over
545  p->left = current->left;
546  }
547 
548  //while loop completed, current = bottomed right node
549  else
550  {
551  //move bottomed right node up one level to take over
552  trailCurrent->right = current->left;
553  }
554 
555  delete current;
556  }
557 }
